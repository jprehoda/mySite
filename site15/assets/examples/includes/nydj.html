ABOUT

"Not your daughter's jeans" was the first project I was presented with @ Chico's. The request was to create a landing page that features a Youtube player with its own built in navigation. I learned a lot about YouTube iframe API during this project. My only regret was not making it responsive, which was not part of the project scope for site projects back then. Please note the styling may look off as the site has gone through several redesigns since the launch of this page.

var style_slider_settings;
var donate_slider_settings;
var isAnimated = false;
// INITIALIZE ON PAGE LOAD
$(document).ready(function(){

	setTimeout( function(){ 
		$(window).resize();
	}, 500 );

	//CHECK FOR STICKY ON SCROLL
	$(window).scroll(shopAllSticky);
	$(window).scroll(backToTopSticky);

	//START STYLES SLIDER
	$(function() {
		$('#style-slide').ulslide({
			effect: {
				type: 'slide', // slide or fade
				axis: 'x',     // x, y
				distance: 0   // Distance between frames
			},
			duration: 500,
			autoslide: 5000,
			width: '100%',
			height: 768,
			pager: '#slide-pager a',
			onAnimateStart: function(settings){
				style_slider_settings = settings;
			},
			onAnimateEnd: function(settings){
				isAnimated = false;
			}
		});
	});

	//STYLE SLIDE SWIPE FUNCTIONALITY
	$("#style-slide").touchwipe({
		wipeLeft: function(e) {
			e.preventDefault();
			style_slider_settings.gotoNext();
			style_slider_settings.autoslide = false;
			$('#style-slide')[0].clearAutoslide();
		},
		wipeRight: function(e) {
			e.preventDefault();
			style_slider_settings.gotoPrev();
			style_slider_settings.autoslide = false;
			$('#style-slide')[0].clearAutoslide();
		},
		min_move_x: 20,
		min_move_y: 20,
		preventDefaultEvents: false
	});
});

function nextSlide(sliderName) {
	sliderName.gotoNext();
}

function prevSlide(sliderName) {
	sliderName.gotoPrev();	
}

// STICKY SHOP ALL
function shopAllSticky(){
	if($(window).scrollTop() > 150) {
		$("#shopAll-cta").css({position:"fixed"});
		if (!$("#shopAll-cta").hasClass('shopAll')){
			$("#shopAll-cta").addClass('shopAll');
		}
	} else {
		$("#shopAll-cta").css({position:"absolute"});
		if ($("#shopAll-cta").hasClass('shopAll')){
			$("#shopAll-cta").removeClass('shopAll');
		}
	}
}

// STICKY BACK TO TOP
function backToTopSticky(){
	if($(window).scrollTop() > 270) {
		$("#back-to-top").css('position', 'fixed');
	} else {
		$("#back-to-top").css('position', 'absolute');
	}
}

$(function() {
	var thumbIndex = 0;
	var activeIndex = 0;
	var projectDate = '20140728';
	var enteredVideoNav = false;
	// SMOOTH SCROLL
	$('.smoothScroll').click(function() {
		if (location.pathname.replace(/^\//,'') == this.pathname.replace(/^\//,'') && location.hostname == this.hostname) {
			var target = $(this.hash);
			target = target.length ? target : $('[name=' + this.hash.slice(1) +']');
			if (target.length) {
				$('html,body').animate({
					scrollTop: target.offset().top
				}, 1000);
				return false;
			}
		}
	});

	//ICID 
	var icidSetter = function(icid){
		var s=s_gi('chchicoscomprod'); 
		s.tl(this,'o',icid);
	}

	// STOP STYLE SLIDER AUTOSLIDE
	$('#slide-pager a').click(function() {
		if(!isAnimated && style_slider_settings.autoslide != false){
			style_slider_settings.autoslide = false;
			$('#style-slide')[0].clearAutoslide();
		}
	});

	// STOP STYLE SLIDER AUTOSLIDE
	$('#style-arrows li').click(function() {
		if(!isAnimated && style_slider_settings.autoslide != false){
			style_slider_settings.autoslide = false;
			$('#style-slide')[0].clearAutoslide();
		}
		if(!isAnimated){
			isAnimated = true;
			if($(this).attr('id') === 'slide-left'){
				prevSlide(style_slider_settings)
				icidSetter('LP-Best-Jeans-Style-Slide-Left');
			} else {
				nextSlide(style_slider_settings)
				icidSetter('LP-Best-Jeans-Style-Slide-Right');
			}
		}
	});

	// PINTEREST CLICK
	$(".pinterest").click(sm_click);

	// VIDEO THUMBNAIL HOVER
	$('.video-thumb').hover(
		//MOUSE OVER
		function() {
			if (!$(this).hasClass("active-thumb")) {
				thumbIndex = $(".video-thumb").index(this);
				activeIndex = $("#video-nav a").index($(".active-thumb"));
				$(this).append("");
				$("#video_description img:nth-child("+(thumbIndex+1)+")").addClass("beta");
				$(".beta").fadeIn("fast");
				$("#video_description img:nth-child("+(activeIndex+1)+")").css("display", "none");
				$(".play-thumb").fadeIn("fast");
				enteredVideoNav = true;
			} else {
				if (enteredVideoNav){
					$("#video_description img:nth-child("+(activeIndex+1)+")").fadeIn("fast");
				}
			}
		},
		//MOUSE OUT
		function() {
			if (!$(this).hasClass("active-thumb")) {
				$(".beta").css("display", "none");
				$(".play-thumb").fadeOut("fast");
				$("#video_description img").removeClass("beta");
				$(".play-thumb").remove();
			}
		}
	);

	//WHEN THE MOUSE LEAVES THE VIDEO NAV, RETURN THE ACTIVE VIDEO DESCRIPTION TO VISIBLE
	$("#video-nav").mouseleave(function() {
		$("#video_description img:nth-child("+(activeIndex+1)+")").fadeIn("fast");
	});

	//CHANGE VIDEO WHEN THUMBNAIL CLICKED
	$('.video-thumb').click(function() {
		if (!$(this).hasClass("active-thumb")) {
			//HIDE THE ACTIVE VIDEO DESCRIPTION
			$("#video_description img:nth-child("+(activeIndex+1)+")").css("display", "none");

			//REMOVE THE ACTIVE THUMBNAIL
			$(".video-thumb span").remove();
			$(".video-thumb").removeClass('active-thumb');

			//REMOVE THE HOVER THUMBNAIL
			$(".play-thumb").remove();

			//ADD THE ACTIVE THUMBNAIL TO THE ELEMENT JUST CLICKED
			$(this).append(' ');
			$(this).addClass('active-thumb');

			//UPDATE ACTIVE INDEX
			activeIndex = $(".video-thumb").index(this);

			//REMOVE THE SECONDARY IMAGE DESCRIPTION
			$("#video_description img").removeClass("beta");

			//GET NEW VIDEO
			getNewVideo(denimVideos[activeIndex]);	
		}
	});
});								}
